Open Microsoft Visual Studio
Create a new project & choose ASP.NET Web Application (.NET Framework)
Project Name → WebApplication1

Right click on project (WebApplication1)→Add→New Item
Select:Web Service (ASMX)→Name:WebService1.asmx

WebService1.asmx.cs:
using System;
using System.Web.Services;

namespace WebApplication1
{
    [WebService(Namespace = "http://tempuri.org/")]
    public class WebService1 : WebService
    {
        [WebMethod]
        public double CelsiusToFahrenheit(double c)
        {
            return (c * 9 / 5) + 32;
        }

        [WebMethod]
        public double FahrenheitToCelsius(double f)
        {
            return (f - 32) * 5 / 9;
        }
    }
}


WebForm1.aspx (design)
Click: Bottom → Design View


WebForm1.aspx.cs:
protected void C_F_Click(object sender, EventArgs e)
{
    double c = Convert.ToDouble(TextBox1.Text);

    WebService1 ws = new WebService1();
    double f = ws.CelsiusToFahrenheit(c);

    F.Text = f.ToString() + " °F";
}

protected void F_C_Click(object sender, EventArgs e)
{
    double f = Convert.ToDouble(TextBox2.Text);

    WebService1 ws = new WebService1();
    double c = ws.FahrenheitToCelsius(f);

    C.Text = c.ToString() + " °C";
}
