package com.mycompany.MathService;

import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.QueryParam;

@Path("/math")
public class MathService {

    @GET
    @Path("/calculate")
    public String calculate(@QueryParam("num") int num) {
        int square = num * num;
        int cube = num * num * num;

        return "Number: " + num +
               "\nSquare: " + square +
               "\nCube: " + cube;
    }
}

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class MathClient {

    public static void main(String[] args) {
        try {
            String urlString = "http://localhost:8080/MathService/webresources/math/calculate?num=4";

            URL url = new URL(urlString);
            HttpURLConnection con = (HttpURLConnection) url.openConnection();
            con.setRequestMethod("GET");

            BufferedReader in = new BufferedReader(
                    new InputStreamReader(con.getInputStream()));

            String inputLine;
            StringBuilder response = new StringBuilder();

            while ((inputLine = in.readLine()) != null) {
                response.append(inputLine).append("\n");
            }

            in.close();

            System.out.println("Response from Service:\n" + response);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
