package com.mycompany.MathService;

import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.QueryParam;

@Path("/math")
public class MathService {

    @GET
    @Path("/calculate")
    public String calculate(@QueryParam("num") int num) {
        int square = num * num;
        int cube = num * num * num;

        return "Number: " + num +
               "\nSquare: " + square +
               "\nCube: " + cube;
    }
}
