using System.Web.Http;

namespace TempService.Controllers
{
    public class TempController : ApiController
    {
        [HttpGet]
        [Route("api/temp/c2f")]
        public string CelsiusToFahrenheit(double c)
        {
            double f = (c * 9 / 5) + 32;
            return "Fahrenheit: " + f;
        }

        [HttpGet]
        [Route("api/temp/f2c")]
        public string FahrenheitToCelsius(double f)
        {
            double c = (f - 32) * 5 / 9;
            return "Celsius: " + c;
        }
    }
}
