Controllers → Add → Controller → API Controller (Empty): TemperatureController.cs: 
using Microsoft.AspNetCore.Mvc;

namespace TemperatureService.Controllers
{
    [ApiController]
    [Route("[controller]")]
    public class TemperatureController : ControllerBase
    {
        // GET: /temperature/c2f?celsius=100
        [HttpGet("c2f")]
        public IActionResult CelsiusToFahrenheit(double celsius)
        {
            double fahrenheit = (celsius * 9 / 5) + 32;
            return Ok(new { celsius = celsius, fahrenheit = fahrenheit });
        }

        // GET: /temperature/f2c?fahrenheit=212
        [HttpGet("f2c")]
        public IActionResult FahrenheitToCelsius(double fahrenheit)
        {
            double celsius = (fahrenheit - 32) * 5 / 9;
            return Ok(new { fahrenheit = fahrenheit, celsius = celsius });
        }
    }
}


It will open Swagger like: https://localhost:5001/swagger

Open in browser: https://localhost:7288/Temperature/c2f?celsius=100
https://localhost:7288/Temperature/f2c?fahrenheit=212



client.py:
import requests

url = "https://localhost:7288/Temperature/f2c"

params = {
    "fahrenheit": 212
}

response = requests.get(url, params=params, verify=False)

data = response.json()

print("Fahrenheit:", data["fahrenheit"])
print("Celsius:", data["celsius"])


run in cmd: python client.py
